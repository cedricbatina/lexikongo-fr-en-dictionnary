<script setup>
import { ref, computed, onMounted, watch, onBeforeUnmount } from 'vue'
import { useRuntimeConfig } from '#imports'
import { useI18n } from 'vue-i18n'
import { loadStripe } from '@stripe/stripe-js'

const { t } = useI18n()
const config = useRuntimeConfig()

const stripePublicKey =
  config.public.stripePublishableKey ||
  config.public.stripeKey ||
  config.public.stripePublicKey ||
  ''

const paypalClientId = config.public.paypalClientId || ''

// EI / bank details
const companyName = computed(() => config.public.companyName || '—')
const companySiret = computed(() => config.public.companySiret || '—')
const bankHolder = computed(() => config.public.bankHolder || '—')
const bankIban = computed(() => config.public.bankIban || '—')
const bankBic = computed(() => config.public.bankBic || '—')

const amount = ref(10)
const donorEmail = ref('')
const anonymous = ref(false)

const methods = [
  { id: 'stripe', labelKey: 'contribute.payments.methods.stripe', icon: 'fas fa-credit-card' },
  { id: 'paypal', labelKey: 'contribute.payments.methods.paypal', icon: 'fab fa-paypal' },
  { id: 'bank', labelKey: 'contribute.payments.methods.bank', icon: 'fas fa-university' },
  { id: 'mobile', labelKey: 'contribute.payments.methods.mobile', icon: 'fas fa-mobile-alt' },
]

const selectedMethod = ref('stripe')
const successMessage = ref('')
const globalError = ref('')

const amountValid = computed(() => typeof amount.value === 'number' && amount.value >= 1)
const displayAmount = computed(() => (amountValid.value ? amount.value.toFixed(0) : '…'))

// helpers timing DOM
const waitForEl = (id, timeout = 2500) =>
  new Promise((resolve) => {
    if (typeof document === 'undefined') return resolve(null)
    const start = Date.now()
    const tick = () => {
      const el = document.getElementById(id)
      if (el) return resolve(el)
      if (Date.now() - start > timeout) return resolve(null)
      requestAnimationFrame(tick)
    }
    tick()
  })

const waitForRef = (refObj, timeout = 2500) =>
  new Promise((resolve) => {
    const start = Date.now()
    const tick = () => {
      if (refObj?.value) return resolve(refObj.value)
      if (Date.now() - start > timeout) return resolve(null)
      requestAnimationFrame(tick)
    }
    tick()
  })

// Stripe
const stripe = ref(null)
const stripeElements = ref(null)
const stripeCardElement = ref(null)
const stripeCardContainer = ref(null)
const stripeReady = ref(false)
const stripeLoading = ref(false)
const stripeError = ref('')

let stripeInitOnce = false

const teardownStripe = () => {
  stripeReady.value = false
  if (stripeCardElement.value) {
    try { stripeCardElement.value.unmount() } catch {}
    stripeCardElement.value = null
  }
}

const initStripe = async () => {
  stripeError.value = ''
  stripeReady.value = false

  if (!stripePublicKey) {
    stripeError.value = t('contribute.payments.stripe.misconfigured')
    return
  }

  try {
    if (!stripeInitOnce) {
      stripe.value = await loadStripe(stripePublicKey)
      stripeInitOnce = true
    }
    if (!stripe.value) throw new Error('Stripe init failed')
    if (!stripeElements.value) stripeElements.value = stripe.value.elements()

    teardownStripe()
    stripeCardElement.value = stripeElements.value.create('card')

    const container = await waitForRef(stripeCardContainer, 4000)
    if (!container) throw new Error('Stripe container missing')

    stripeCardElement.value.mount(container)
    stripeReady.value = true
  } catch (err) {
    stripeError.value = err?.message || t('contribute.payments.stripe.initError')
    stripeReady.value = false
  }
}

const onStripeSubmit = async () => {
  if (!stripe.value || !stripeCardElement.value || !amountValid.value) {
    stripeError.value = t('contribute.payments.stripe.missing')
    return
  }

  stripeLoading.value = true
  stripeError.value = ''
  successMessage.value = ''
  globalError.value = ''

  try {
    const { clientSecret } = await $fetch('/api/payments/stripe-intent', {
      method: 'POST',
      body: {
        amount: amount.value,
        anonymous: anonymous.value,
        donorEmail: donorEmail.value || null,
      },
    })

    const { error, paymentIntent } = await stripe.value.confirmCardPayment(clientSecret, {
      payment_method: { card: stripeCardElement.value },
    })

    if (error) {
      stripeError.value = error.message || 'Stripe error'
      return
    }

    if (paymentIntent?.status === 'succeeded') {
      successMessage.value = t('contribute.payments.stripe.success')
    } else {
      globalError.value = t('contribute.payments.stripe.unknownStatus')
    }
  } catch (err) {
    globalError.value = t('contribute.payments.stripe.serverError')
  } finally {
    stripeLoading.value = false
  }
}

// PayPal
const paypalReady = ref(false)
const paypalError = ref('')

let paypalScriptPromise = null
let paypalRendered = false

const teardownPaypal = () => {
  paypalReady.value = false
  paypalRendered = false
  if (typeof document !== 'undefined') {
    const container = document.getElementById('paypal-button-container')
    if (container) container.innerHTML = ''
  }
}

const loadPaypalScript = () => {
  if (paypalScriptPromise) return paypalScriptPromise

  paypalScriptPromise = new Promise((resolve, reject) => {
    if (typeof window === 'undefined') return resolve(false)
    if (window.paypal) return resolve(true)

    const script = document.createElement('script')
    script.src = `https://www.paypal.com/sdk/js?client-id=${paypalClientId}&currency=EUR`
    script.async = true
    script.onload = () => resolve(true)
    script.onerror = () => reject(new Error('paypal sdk load failed'))
    document.head.appendChild(script)
  })

  return paypalScriptPromise
}

const renderPaypalButtons = () => {
  if (!window.paypal) return
  const container = document.getElementById('paypal-button-container')
  if (!container) return

  container.innerHTML = ''

  window.paypal.Buttons({
    createOrder: (data, actions) => {
      const value = amountValid.value ? amount.value.toFixed(2) : '10.00'
      return actions.order.create({
        purchase_units: [{ amount: { value, currency_code: 'EUR' } }],
      })
    },
    onApprove: async (data, actions) => {
      try {
        const details = await actions.order.capture()
        successMessage.value = t('contribute.payments.paypal.success', {
          name: details?.payer?.name?.given_name || '',
        })
        paypalError.value = ''

        await $fetch('/api/payments/paypal-capture', {
          method: 'POST',
          body: {
            orderId: data.orderID,
            raw: details,
            amount: amountValid.value ? amount.value : null,
            currency: 'EUR',
            anonymous: anonymous.value,
            donorEmail: donorEmail.value || details?.payer?.email_address || null,
          },
        })
      } catch (e) {
        paypalError.value = t('contribute.payments.paypal.error')
      }
    },
    onError: (err) => {
      paypalError.value =
        t('contribute.payments.paypal.error') + (err?.message ? ` (${err.message})` : '')
    },
  }).render('#paypal-button-container')

  paypalReady.value = true
  paypalRendered = true
}

const initPaypal = async () => {
  paypalError.value = ''
  paypalReady.value = false

  if (!paypalClientId) {
    paypalError.value = t('contribute.payments.paypal.misconfigured')
    return
  }

  try {
    await loadPaypalScript()
    const el = await waitForEl('paypal-button-container', 4000)
    if (!el) throw new Error('paypal container missing')

    if (!paypalRendered) renderPaypalButtons()
  } catch (err) {
    paypalError.value = t('contribute.payments.paypal.loadError')
    paypalReady.value = false
  }
}

// Clipboard
const copiedText = ref('')
const copyToClipboard = (text, field) => {
  if (!text) return
  if (typeof navigator === 'undefined' || !navigator.clipboard) return

  navigator.clipboard.writeText(String(text)).then(() => {
    const map = {
      IBAN: t('contribute.bank.ibanCopied'),
      BIC: t('contribute.bank.bicCopied'),
      SIRET: t('contribute.bank.siretCopied'),
      HOLDER: t('contribute.bank.holderCopied'),
      BENEFICIARY: t('contribute.bank.beneficiaryCopied'),
    }
    copiedText.value = map[field] || t('contribute.bank.copied')
    setTimeout(() => (copiedText.value = ''), 2000)
  })
}

// Mobile money
const mobileForm = ref({
  name: '',
  phone: '',
  operator: 'MTN',
})

const mobileLoading = ref(false)
const mobileError = ref('')

const onMobileMoneySubmit = async () => {
  mobileError.value = ''
  successMessage.value = ''
  globalError.value = ''

  if (!mobileForm.value.name || !mobileForm.value.phone) {
    mobileError.value = t('contribute.mobile.missingFields')
    return
  }

  try {
    mobileLoading.value = true

    await $fetch('/api/contribute/mobile-money-request', {
      method: 'POST',
      body: {
        ...mobileForm.value,
        country: 'Congo',
        suggestedAmount: amountValid.value ? amount.value : null,
        anonymous: anonymous.value,
        donorEmail: donorEmail.value || null,
      },
    })

    successMessage.value = t('contribute.mobile.success')
    mobileForm.value = { name: '', phone: '', operator: 'MTN' }
  } catch (err) {
    mobileError.value = t('contribute.mobile.error')
  } finally {
    mobileLoading.value = false
  }
}

// select method
const onSelectMethod = (methodId) => {
  if (selectedMethod.value === methodId) return

  if (selectedMethod.value === 'stripe') teardownStripe()
  if (selectedMethod.value === 'paypal') teardownPaypal()

  selectedMethod.value = methodId
  successMessage.value = ''
  globalError.value = ''
  stripeError.value = ''
  paypalError.value = ''
}

watch(
  () => selectedMethod.value,
  async (method) => {
    if (method === 'stripe') await initStripe()
    if (method === 'paypal') await initPaypal()
  }
)

onMounted(async () => {
  if (selectedMethod.value === 'stripe') await initStripe()
})

onBeforeUnmount(() => {
  teardownStripe()
  teardownPaypal()
})
</script>
